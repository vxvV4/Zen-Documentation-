Zen Library V 2.A0

By VantaXock

Installation

local library = loadstring(game:HttpGet("URL"))() library:Init()

Create Window

local window = library:CreateWindow("Window Name")

Components

Button

window:Button("Button Name", function() print("clicked") end)

Toggle

window:Toggle("Toggle Name", function(enabled) print("Toggle:", enabled) end)

TextBox

window:Box("TextBox Name", function(text) print("Text:", text) end)

Keybind

window:Keybind("Keybind Name", Enum.KeyCode.F, "toggle", function(active) print("Keybind:", active) end)

Dropdown

window:Dropdown("Dropdown Name", {"Option 1", "Option 2"}, function(selected) print("Selected:", selected) end)

Slider

window:Slider("Slider Name", 0, 100, false, function(value) print("Value:", value) end)

Folder

local folder = window:AddFolder("Folder Name") -- All components work inside folders folder:Button("Folder Button", function() print("folder button") end) folder:Toggle("Folder Toggle", function(enabled) print("Folder toggle:", enabled) end) folder:Box("Folder TextBox", function(text) print("Folder text:", text) end) folder:Keybind("Folder Keybind", Enum.KeyCode.G, "hold", function(active) print("Folder keybind:", active) end) folder:Dropdown("Folder Dropdown", {"Folder Option 1", "Folder Option 2"}, function(selected) print("Folder selected:", selected) end) folder:Slider("Folder Slider", 1, 50, false, function(value) print("Folder value:", value) end)

Initialize

library:Init()

Complete Usage

-- Load and initialize the library local library = loadstring(game:HttpGet("URL"))() -- Create main window local window = library:CreateWindow("My GUI") -- Add a simple button window:Button("Test", function() print("test") end) -- Add a toggle switch window:Toggle("Feature", function(enabled) if enabled then print("on") else print("off") end end) -- Add a text input box window:Box("Enter Text", function(text) print("You entered:", text) end) -- Add a keybind (F key to toggle) window:Keybind("Toggle GUI", Enum.KeyCode.F, "toggle", function(active) print("GUI toggled:", active) end) -- Add a dropdown menu window:Dropdown("Select Option", {"Option 1", "Option 2", "Option 3"}, function(selected) print("Selected:", selected) end) -- Add a slider (0 to 100) window:Slider("Value", 0, 100, false, function(value) print("Slider value:", value) end) -- Create a folder for organization local folder = window:AddFolder("Advanced") -- Add all component types to the folder folder:Button("Folder Button", function() print("folder button clicked") end) folder:Toggle("Folder Feature", function(enabled) print("folder toggle:", enabled) end) folder:Box("Folder Input", function(text) print("folder input:", text) end) folder:Keybind("Folder Hotkey", Enum.KeyCode.H, "toggle", function(active) print("folder keybind:", active) end) folder:Dropdown("Folder Menu", {"Folder A", "Folder B"}, function(selected) print("folder dropdown:", selected) end) folder:Slider("Folder Range", 10, 90, false, function(value) print("folder slider:", value) end) library:Init()
