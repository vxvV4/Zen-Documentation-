Zen Library V 2.A0

By VantaXock

Installation

local library = loadstring(game:HttpGet("URL"))() library:Init()

Create Window

local window = library:CreateWindow("Window Name")

Components

Button

window:Button("Button Name", function() print("clicked") end)

Toggle

window:Toggle("Toggle Name", function(enabled) print("Toggle:", enabled) end)

TextBox

window:Box("TextBox Name", function(text) print("Text:", text) end)

Keybind

window:Keybind("Keybind Name", Enum.KeyCode.F, "toggle", function(active) print("Keybind:", active) end)

Dropdown

window:Dropdown("Dropdown Name", {"Option 1", "Option 2"}, function(selected) print("Selected:", selected) end)

Slider

window:Slider("Slider Name", 0, 100, false, function(value) print("Value:", value) end)

Folder

local folder = window:AddFolder("Folder Name") folder:Button("Button in Folder", function() print("folder button") end)

Initialize

library:Init()

Complete Usage

-- Load and initialize the library local library = loadstring(game:HttpGet("URL"))() -- Create main window local window = library:CreateWindow("My GUI") -- Add a simple button window:Button("Test", function() print("test") end) -- Add a toggle switch window:Toggle("Feature", function(enabled) if enabled then print("on") else print("off") end end) -- Add a text input box window:Box("Enter Text", function(text) print("You entered:", text) end) -- Add a keybind (F key to toggle) window:Keybind("Toggle GUI", Enum.KeyCode.F, "toggle", function(active) print("GUI toggled:", active) end) -- Add a dropdown menu window:Dropdown("Select Option", {"Option 1", "Option 2", "Option 3"}, function(selected) print("Selected:", selected) end) -- Add a slider (0 to 100) window:Slider("Value", 0, 100, false, function(value) print("Slider value:", value) end) -- Create a folder for organization local folder = window:AddFolder("Advanced") folder:Button("Folder Button", function() print("folder button clicked") end) library:Init()
